% PDF: light gray code panels + syntax colors. Token macros exist only when the
% document has highlighted code blocks.
\usepackage{etoolbox}

% Block quotes (Markdown `>`): keep this dependency-light so PDF output works
% with smaller TeX installs.
\usepackage{xcolor}
\renewenvironment{quote}{%
  \list{}{%
    \leftmargin=1.4em%
    \rightmargin=0.6em%
    \parsep=0.35em%
    \topsep=0.65em%
  }%
  \item\relax%
  \begingroup%
  \color{black!82}%
}{%
  \endgroup%
  \endlist%
}
