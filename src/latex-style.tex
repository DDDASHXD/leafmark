% PDF: light gray code panels + syntax colors. Token macros exist only when the
% document has highlighted code blocks.
\usepackage{etoolbox}

% Block quotes (Markdown `>`): Pandoc uses the `quote` environment - add a left
% bar and light fill so they read as distinct from body text.
\usepackage{xcolor}
\usepackage{mdframed}
\renewenvironment{quote}{%
  \begin{mdframed}[
    hidealllines=true,
    leftline=true,
    linewidth=2.4pt,
    linecolor=blue!50!black,
    backgroundcolor=gray!8,
    innerleftmargin=0.9em,
    innerrightmargin=0.9em,
    innertopmargin=0.55em,
    innerbottommargin=0.55em,
    skipabove=\smallskipamount,
    skipbelow=\smallskipamount
  ]%
}{%
  \end{mdframed}%
}
